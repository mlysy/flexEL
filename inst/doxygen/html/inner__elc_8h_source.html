163 inline double flexEL::InnerELC::LogSharp(double x, double q) {
164  double xq;
165  if(x >= q) {
166  return(log(x));
167  } else {
168  xq = x/q;
169  return(-0.5*xq*xq + 2.0*xq - 1.5 + log(q));
170  }
171 }

443 inline void flexEL::InnerELC::LambdaNR(int& n_iter, double& max_err) {
444 
445  lambda_old_ = lambda0_;
446  lambda_new_.fill(0.0);
447 
448  // Note: these two cannot be preallocate untill `supp` is set
449  // but has to do it this way since .block does not return a MatrixXd (TODO: alternative way?)
450  GGt_used_ = GGt_.block(0,0,n_eqs_,n_obs2_*n_eqs_);
451  G_used_ = G_.block(0,0,n_eqs_,n_obs2_);
452 
453  block_outer(GGt_used_,G_used_);
454 
455  // To avoid numerical problem (TODO: test this again and see if it is ok without it?)
456  VectorXd weights_save = weights_;
457  for (int ll=0;ll<n_obs2_; ll++) {
458  if (weights_save(ll) < rel_tol_*n_obs_) weights_save(ll) = 0;
459  }
460 
461  // newton-raphson loop
462  int ii;
463  for(ii=0; ii<max_iter_; ii++) {
464  // Q1 and Q2
465  Glambda_.noalias() = lambda_old_.transpose() * G_used_;
466  Glambda_ = weights_.head(n_obs2_).sum() + Glambda_.array();
467  Q2_.fill(0.0);
468  for(int jj=0; jj<n_obs2_; jj++) {
469  rho_(jj) = LogSharp1(Glambda_(jj), weights_(jj));
470 
471  // To avoid numerical problem
472  if (weights_(jj) > rel_tol_*n_obs_) {
473  Q2_ += weights_(jj) * LogSharp2(Glambda_(jj), weights_(jj)) * GGt_used_.block(0,jj*n_eqs_,n_eqs_,n_eqs_);
474  }
475  }
476  // To avoid numerical problem
477  for (int kk=0; kk<n_obs2_; kk++) {
478  if (isinf(rho_(kk)) && weights_save(kk) == 0) rho_(kk) = 0;
479  }
480  // update lambda
481  Q1_ = G_used_ * (rho_.array()*weights_save.head(n_obs2_).array()).matrix();
482  Q2ldlt_.compute(Q2_);
483  lambda_new_.noalias() = lambda_old_ - Q2ldlt_.solve(Q1_);
484  max_err = MaxRelErr();
485  if (max_err < rel_tol_) {
486  break;
487  }
488  lambda_old_ = lambda_new_; // complete cycle
489  }
490  n_iter = ii; // output lambda and also n_iter and max_err
491  return;
492 }
