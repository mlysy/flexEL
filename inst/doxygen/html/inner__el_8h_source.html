179 inline flexEL::InnerEL::InnerEL(int n_obs, int n_eqs) {
180  // assign internal values
181  n_obs_ = n_obs;
182  n_eqs_ = n_eqs;
183  // support correction
184  supp_ = false;
185  n_obs1_ = n_obs_+1;
186  n_obs2_ = n_obs_+supp_;
187  // initialization of constants
188  trunc_ = 1.0 / n_obs_;
189  aa_ = -.5 * n_obs_*n_obs_;
190  bb_ = 2.0 * n_obs_;
191  cc_ = -1.5 - log(n_obs_);
192  // space allocation
193  omegas_ = VectorXd::Zero(n_obs1_).array() + 1.0/(double)n_obs1_; // Initialize to 1/n_obs_
194  G_ = MatrixXd::Zero(n_eqs_,n_obs1_); // augmented G
195  GGt_ = MatrixXd::Zero(n_eqs_,n_obs1_*n_eqs_);
196  lambda0_ = VectorXd::Zero(n_eqs_); // default initial value of lambda
197  lambda_old_ = VectorXd::Zero(n_eqs_); // Initialize to all 0's
198  lambda_new_ = VectorXd::Zero(n_eqs_);
199  Q1_ = VectorXd::Zero(n_eqs_);
200  Q2_ = MatrixXd::Zero(n_eqs_,n_eqs_);
201  Glambda_ = VectorXd::Zero(n_obs1_);
202  Gl11_ = ArrayXd::Zero(n_obs1_);
203  rho_ = VectorXd::Zero(n_obs1_);
204  rel_err_ = VectorXd::Zero(n_eqs_);
205  Q2ldlt_.compute(MatrixXd::Identity(n_eqs_,n_eqs_));

310 inline void flexEL::InnerEL::LambdaNR(int& n_iter, double& max_iter) {
311 
312  lambda_old_ = lambda0_; // set to initial value
313  lambda_new_.fill(0.0); // may not be needed here..
314 
315  // Note: these two cannot be preallocate untill `supp` is set
316  GGt_used_ = GGt_.block(0,0,n_eqs_,n_obs2_*n_eqs_);
317  G_used_ = G_.block(0,0,n_eqs_,n_obs2_);
318 
319  block_outer(GGt_used_,G_used_);
320 
321  // newton-raphson loop
322  int ii, jj;
323  for(ii=0; ii<max_iter_; ii++) {
324  // Q1 and Q2
325  Glambda_.noalias() = lambda_old_.transpose() * G_used_;
326  Glambda_= 1.0 - Glambda_.array();
327  Q2_.fill(0.0);
328  for(jj=0; jj<n_obs2_; jj++) {
329  rho_(jj) = LogStar1(Glambda_(jj));
330  Q2_ += LogStar2(Glambda_(jj)) * GGt_used_.block(0,jj*n_eqs_,n_eqs_,n_eqs_);
331  }
332  // update lambda
333  Q1_ = -G_used_ * rho_;
334  Q2ldlt_.compute(Q2_);
335  lambda_new_.noalias() = lambda_old_ - Q2ldlt_.solve(Q1_);
336  max_iter = MaxRelErr();
337  if (max_iter < rel_tol_) {
338  break;
339  }
340  lambda_old_ = lambda_new_; // complete cycle
341  }
342  n_iter = ii; // output lambda and also n_iter and max_iter
343  return;
344 }

379 inline void flexEL::InnerEL::LogELGrad(double& logel, MatrixXd& dldG) {
380  if (omegas_.head(n_obs2_) != omegas_.head(n_obs2_)) {
381  logel = -INFINITY;
382  dldG = MatrixXd::Zero(n_eqs_,n_obs2_);
383  }
384  logel = omegas_.head(n_obs2_).array().log().sum();
385  if (supp_ == false) {
386  dldG = n_obs_ * lambda_new_ * omegas_.head(n_obs_).transpose();
387  }
388  else {
389  dldG = n_obs2_ * lambda_new_ * omegas_.head(n_obs_).transpose() -
390  n_obs2_ * omegas_(n_obs2_-1) * supp_a_/n_obs_ * lambda_new_ * VectorXd::Ones(n_obs_).transpose();
391  }
