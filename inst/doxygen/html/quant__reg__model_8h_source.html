246 inline void flexEL::QuantRegModel::EvalG(Ref<MatrixXd> G, const Ref<const MatrixXd>& Beta) {
247  // stack multiple "Gs" for each quantile level
248  for(int jj=0; jj<n_qts_; jj++) {
249  for(int ii=0; ii<y_.size(); ii++) {
250  G.block(jj*n_bet_,ii,n_bet_,1) = phi_tau(y_(ii)-X_.col(ii).transpose()*Beta.col(jj), tau[jj])*X_.col(ii);
251  }
252  }

266 inline void flexEL::QuantRegModel::EvalG(Ref<MatrixXd> G,
267  const Ref<const VectorXd>& beta,
268  const Ref<const VectorXd>& gamma,
269  const double& sig2,
270  const Ref<const VectorXd>& nu) {
271  // TODO: warning if negative sig2?
272  // if (sig2 < 0) std::cout << "qrls.EvalG: negative variance." << std::endl;
273  eZg_.array() = (-gamma.transpose()*Z_).array().exp();
274  yXbeZg_.array() = (y_.transpose()-beta.transpose()*X_).array() * eZg_.array();
275  yXbeZg2_.array() = yXbeZg_.array()*yXbeZg_.array();
276 
277  tG_ = MatrixXd::Zero(n_obs_,n_eqs_); // NEW: DEC 25
278  // 1st deriv w.r.t beta
279  tG_.block(0,0,n_obs_,n_bet_) = X_.transpose();
280  tG_.block(0,0,n_obs_,n_bet_).array().colwise() *= yXbeZg_.transpose().array() * eZg_.transpose().array();
281  // 1st deriv w.r.t gamma
282  tG_.block(0,n_bet_,n_obs_,n_gam_) = Z_.transpose();
283  tG_.block(0,n_bet_,n_obs_,n_gam_).array().colwise() *= (1.0-yXbeZg2_.transpose().array()/sig2); // NEW (DEC 6)
284  // variance param
285  tG_.block(0,n_bet_+n_gam_,n_obs_,1).array() = 1/sig2*yXbeZg2_.transpose().array()-1;
286  // quantile param(s)
287  for (int ii=0; ii<n_obs_; ii++) {
288  for (int jj=0; jj<n_qts_; jj++) {
289  tG_.block(ii,n_bet_+n_gam_+1+jj,1,1).array() = phi_tau(yXbeZg_(ii)/sqrt(sig2)-nu(jj), tau[jj]);
290  }
291  }
