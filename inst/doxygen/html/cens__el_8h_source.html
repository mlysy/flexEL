223  inline void CensEL::eval_pso(double &psos,
224  const int ii,
225  const Ref<const VectorXd>& omega) {
226  psos = 0;
227  int kk;
228  for (int jj=n_obs2_-1; jj>=0; jj--) {
229  // Note: eps_ord_ corresponds to epsilons in ascending order
230  kk = eps_ord_(jj); // kk is the index of the jj-th largest epsilon
231  psos += omega(kk);
232  if (kk == ii) break; // until (backwardly) reaching ii-th largest epsilon
233  }
234  return;
235  }

241  // std::cout << "epsilon = " << epsilon.transpose() << std::endl;

288  const Ref<const VectorXd>& delta,

296  for (int ii=0; ii<n_obs2_; ii++) {
297  eval_pso_smooth(psos_vec_(ii), ii, omega, epsilon_.head(n_obs2_));
298  }
299  // std::cout << "psos_vec_ = " << psos_vec_.transpose() << std::endl;
300 
301  if (supp_adj_) {
302  for (int jj=0; jj<n_obs2_; jj++) {
303  // std::cout << "jj = " << jj << std::endl;
304  for (int kk=0; kk<n_obs2_; kk++) {
305  // std::cout << " kk = " << kk << std::endl;
306  if (jj == n_obs2_-1 && kk == n_obs2_-1) {
307  psot_(jj) += (1-delta_(kk))*ind_smooth(0.0,smooth_s_)*omega(jj)/psos_vec_(kk);
308  }
309  else {
310  psot_(jj) += (1-delta_(kk))*ind_smooth(epsilon_(kk)-epsilon_(jj),smooth_s_)*omega(jj)/psos_vec_(kk);
311  }
312  }
313  }
314  }
315  else {
316  for (int jj=0; jj<n_obs2_; jj++) {
317  for (int kk=0; kk<n_obs2_; kk++) {
318  psot_(jj) += (1-delta_(kk))*ind_smooth(epsilon_(kk)-epsilon_(jj),smooth_s_)*omega(jj)/psos_vec_(kk);
319  }
320  }
321  }
322  // std::cout << "psot_ = " << psot_.transpose() << std::endl;
323  weights = delta_.array() + psot_.array();

329  inline void CensEL::omega_hat(Ref<VectorXd> omega,
330  const Ref<const MatrixXd>& G,
331  const Ref<const VectorXd>& delta,
332  const Ref<const VectorXd>& epsilon) {
333  // std::cout << "omega_hat: omega = " << omega.transpose() << std::endl;
334  // Need to have a valid starting value if the last one is not valid in MCMC
335  if (omega != omega) { // if there is nan in omega
336  // std::cout << "omega_hat: resetting omega_." << std::endl;
337  omega = omega_init_;
338  }
339  int em_iter;
340  double em_err;
341  VectorXd weights(n_obs2_);
342  if (!smooth_) {
343  eval_weights(weights, delta, epsilon, omega);
344  // std::cout << "delta = " << omega.transpose() << std::endl;
345  // std::cout << "epsilon = " << omega.transpose() << std::endl;
346  // std::cout << "initial omega = " << omega.transpose() << std::endl;
347  // std::cout << "initial weights = " << weights.transpose() << std::endl;
348  } else{
349  eval_weights_smooth(weights, delta, epsilon, omega);
350  }
351  double sum_weights = weights.head(n_obs2_).sum();
352  norm_weights_.head(n_obs2_) = weights/sum_weights;
353  // std::cout << "initial norm_weights_.head(n_obs2_) = " << norm_weights_.head(n_obs2_).transpose() << std::endl;
354  double logel_old = GEL.logel_omega(omega, norm_weights_.head(n_obs2_), sum_weights);
355  double logel = logel_old;
356  // std::cout << "initial logel = " << logel << std::endl;
357  int ii;
358  VectorXd lambda(n_eqs_);
359  // std::cout << "before EM loop" << std::endl;
360  // std::cout << "norm_weights_ = " << norm_weights_.transpose() << std::endl;
361  for(ii=0; ii<max_iter_em_; ii++) {
362  // std::cout << "EM ii = " << ii << std::endl;
363  // E-step:
364  if (!smooth_) {
365  eval_weights(weights, delta, epsilon, omega);
366  } else{
367  eval_weights_smooth(weights, delta, epsilon, omega);
368  }
369  sum_weights = weights.head(n_obs2_).sum();
370  norm_weights_.head(n_obs2_) = weights/sum_weights;
371  // std::cout << "norm_weights_ = " << norm_weights_.transpose() << std::endl;
372  // M-step:
373  GEL.lambda_nr(lambda, G, norm_weights_.head(n_obs2_));
374  GEL.omega_hat(omega, lambda, G, norm_weights_.head(n_obs2_));
375  logel = GEL.logel_omega(omega, norm_weights_.head(n_obs2_), sum_weights);
376  em_err = abs(logel-logel_old); // absolute error in log EL
377  if (em_err < abs_tol_) break;
378  logel_old = logel;
379  }
380  em_iter = ii;
381  if (em_iter == max_iter_em_ && em_err > abs_tol_) {
382  // TODO: maybe should assign nan elsewhere
383  for (int ii=0; ii<n_obs2_; ii++) {
384  omega(ii) = std::numeric_limits<double>::quiet_NaN();
385  }
386  }
387  em_iter_ = em_iter;
388  em_err_ = em_err;
389  return;
390  }
