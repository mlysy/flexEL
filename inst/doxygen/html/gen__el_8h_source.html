278  inline void GenEL::lambda_nr_impl(Ref<VectorXd> lambda,
279  const Ref<const MatrixXd>& G,
280  const Ref<const VectorXd>& norm_weights) {
281  int n_obs = G.cols();
282  lambda = lambda0_; // set to initial value
283  // logstar constants
284  // assume the weights are normalized to sum to one
285  bstar_ = norm_weights.array().inverse();
286  astar_ = -.5 * bstar_.array().square();
287  bstar_ *= 2.0;
288  // newton-raphson loop
289  int ii, jj;
290  for(ii=0; ii<max_iter_; ii++) {
291  // Q1 and Q2
292  Glambda_.noalias() = lambda.transpose() * G;
293  Glambda_= 1.0 - Glambda_.array();
294  Q2_.fill(0.0);
295  for(jj=0; jj<n_obs; jj++) {
296  rho_(jj) = log_star1(Glambda_(jj),
297  norm_weights(jj), astar_(jj), bstar_(jj));
298  GGt_.noalias() = G.col(jj) * G.col(jj).transpose();
299  Q2_ -= norm_weights(jj) *
300  log_star2(Glambda_(jj), norm_weights(jj), astar_(jj)) * GGt_;
301  }
302  // update lambda
303  rho_.array() *= norm_weights.array();
304  Q1_.noalias() = G * rho_;
305  Q2llt_.compute(Q2_);
306  Q2llt_.solveInPlace(Q1_);
307  lambda_new_ = lambda - Q1_;
308  nr_err_ = max_rel_err(lambda, lambda_new_);
309  lambda = lambda_new_; // complete cycle
310  if (nr_err_ < rel_tol_) {
311  break;
312  }
313  }
314  nr_iter_ = ii;
315  return;
316  }
