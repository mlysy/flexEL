88 inline void MwgAdapt::adapt(double *mwgSd, bool *accept) {
89  nIter++; // add one iteration
90  for(int ii=0; ii<nRV; ii++) {
91  if(doAdapt[ii]) {
92  nAccept[ii] += (int) accept[ii];
93  acc = (double) nAccept[ii] / (double) nIter;
94  delta = pow((double) nIter, -adaptRate[ii]);
95  // // DEBUG BEGINS
96  // // std::cout << "nIter = " << nIter << std::endl;
97  // if (nIter % 200 == 0) std::cout << "delta = " << delta << std::endl;
98  // // DEBUG ENDS
99  if(delta > adaptMax[ii]) delta = adaptMax[ii];
100  lsig = log(mwgSd[ii]);
101  lsig += acc < targAcc ? -delta : delta;
102  mwgSd[ii] = exp(lsig);
103  }
104  }
105  // return;
106 }
