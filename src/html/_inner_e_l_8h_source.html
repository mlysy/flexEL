327 template<typename ELModel>
328 inline void el::InnerEL<ELModel>::lambdaNR(int& nIter, double& maxErr) {
329 
330  lambdaOld_ = lambda0_; // set to initial value
331  lambdaNew_.fill(0.0); // may not be needed here..
332 
333  // Note: these two cannot be preallocate untill `support` is set
334  GGtUsed_ = GGt_.block(0,0,nEqs_,nObs2_*nEqs_);
335  GUsed_ = G_.block(0,0,nEqs_,nObs2_);
336 
337  block_outer(GGtUsed_,GUsed_);
338 
339  // newton-raphson loop
340  int ii, jj;
341  for(ii=0; ii<maxIter_; ii++) {
342  // Q1 and Q2
343  Glambda_.noalias() = lambdaOld_.transpose() * GUsed_;
344  Glambda_= 1.0 - Glambda_.array();
345  Q2_.fill(0.0);
346  for(jj=0; jj<nObs2_; jj++) {
347  rho_(jj) = logstar1(Glambda_(jj));
348  Q2_ += logstar2(Glambda_(jj)) * GGtUsed_.block(0,jj*nEqs_,nEqs_,nEqs_);
349  }
350  // update lambda
351  Q1_ = -GUsed_ * rho_;
352  Q2ldlt_.compute(Q2_);
353  lambdaNew_.noalias() = lambdaOld_ - Q2ldlt_.solve(Q1_);
354  maxErr = maxRelErr();
355  if (maxErr < relTol_) {
356  break;
357  }
358  lambdaOld_ = lambdaNew_; // complete cycle
359  }
360  nIter = ii; // output lambda and also nIter and maxErr
361  return;
362 }
