253  tG_.block(0,0,nObs_,nBet_).array().colwise() *= yXbeZg_.transpose().array() * eZg_.transpose().array();
254  // 1st deriv w.r.t gamma
255  tG_.block(0,nBet_,nObs_,nGam_) = Z_.transpose();
256  // tG.block(0,nBet,nObs,nGam).array().colwise() *= yXbeZg2.transpose().array();

258  tG_.block(0,nBet_,nObs_,nGam_).array().colwise() *= (1.0-yXbeZg2_.transpose().array()/sig2); // NEW (DEC 6)
259  // variance param

266  }
