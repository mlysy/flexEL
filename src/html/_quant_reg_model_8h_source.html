152 inline void QuantRegModel::evalG(const Ref<const VectorXd>& beta,
153  const Ref<const VectorXd>& gamma,
154  const double& sig2,
155  const Ref<const VectorXd>& Nu) {
156  // std::cout << "nQts = " << nQts << std::endl;
157  // TODO: warning if negative sig2?
158  if (sig2 < 0) std::cout << "qrls.evalG: negative variance." << std::endl;
159  eZg.array() = (-gamma.transpose()*Z).array().exp();
160  yXbeZg.array() = (y.transpose()-beta.transpose()*X).array() * eZg.array();
161  yXbeZg2.array() = yXbeZg.array()*yXbeZg.array();
162  // 1st deriv w.r.t beta
163  tG.block(0,0,nObs,nBet) = X.transpose();
164  tG.block(0,0,nObs,nBet).array().colwise() *= yXbeZg.transpose().array() * eZg.transpose().array();
165  // 1st deriv w.r.t gamma
166  tG.block(0,nBet,nObs,nGam) = Z.transpose();
167  // tG.block(0,nBet,nObs,nGam).array().colwise() *= yXbeZg2.transpose().array();
168  tG.block(0,nBet,nObs,nGam).array().colwise() *= (1.0-yXbeZg2.transpose().array());
169  // variance param
170  tG.block(0,nBet+nGam,nObs,1).array() = 1/sig2*yXbeZg2.transpose().array()-1;
171  // std::cout << "tG = \n" << tG << std::endl;
172  // quantile param(s)
173  for (int ii=0; ii<nObs; ii++) {
174  for (int jj=0; jj<nQts; jj++) {
175  tG.block(ii,nBet+nGam+1+jj,1,1).array() = phi_alpha(yXbeZg(ii)/sqrt(sig2)-Nu(jj), alpha[jj]);
176  }
177  }
178  G = tG.transpose();
179 }
