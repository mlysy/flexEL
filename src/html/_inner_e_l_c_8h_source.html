236 inline double InnerELC<ELModel>::logsharp(double x, double q) {
237  double xq;
238  if(x >= q) {
239  return(log(x));
240  } else {
241  xq = x/q;
242  return(-0.5*xq*xq + 2.0*xq - 1.5 + log(q));
243  }

305 template<typename ELModel>
306 inline void InnerELC<ELModel>::lambdaNR(int& nIter, double& maxErr) {
307  // prevent bad starting values
308  lambdaNew.fill(0.0);
309  lambdaOld.fill(0.0);
310  // blockOuter(); // initialize GGt according to epsilons order (epsOrd)
311  block_outer(GGt,G);
312  int ii;
313 
314  // TODO: to avoid numerical problem
315  VectorXd weights_save = weights;
316  for (int ll=0;ll<nObs; ll++) {
317  if (weights_save(ll) < relTol*nObs) weights_save(ll) = 0;
318  }
319 
320  // newton-raphson loop
321  for(ii=0; ii<maxIter; ii++) {
322  // std::cout << "lambdaOld = " << lambdaOld.transpose() << std::endl;
323  // Q1 and Q2
324  Glambda.noalias() = lambdaOld.transpose() * G;
325  Glambda = weights.sum() + Glambda.array();
326  Q2.fill(0.0);
327  for(int jj=0; jj<nObs; jj++) {
328  rho(jj) = logsharp1(Glambda(jj), weights(jj));
329 
330  // TODO: to avoid numerical problem
331  if (weights(jj) > relTol*nObs) {
332  Q2 += weights(jj) * logsharp2(Glambda(jj), weights(jj)) * GGt.block(0,jj*nEqs,nEqs,nEqs);
333  }
334  }
335 
336  // TODO: to avoid numerical problem
337  for (int kk=0; kk<nObs; kk++) {
338  if (isinf(rho(kk)) && weights_save(kk) == 0) rho(kk) = 0;
339  }
340 
341  Q1 = G * (rho.array()*weights_save.array()).matrix();
342  // Q1 = G * (rho.array()*weights.array()).matrix();

371 template<typename ELModel>
372 inline double InnerELC<ELModel>::evalPsos(const int ii) {
373  double psos = 0;
374  int kk;
375  for (int jj=nObs-1; jj>=0; jj--) {
376  // epsOrd corresponds to epsilons acesndingly
377  kk = epsOrd(jj); // kk is the index of the jj-th largest epsilon
378  psos += omegas(kk);
379  if (kk == ii) break; // until (backwardsly) reaching ii-th largest epsilon
380  }
381  return psos;
382 }
