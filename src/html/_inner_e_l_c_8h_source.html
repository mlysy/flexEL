389 inline double el::InnerELC<ELModel>::logsharp(double x, double q) {
390  double xq;
391  if(x >= q) {
392  return(log(x));
393  } else {
394  xq = x/q;
395  return(-0.5*xq*xq + 2.0*xq - 1.5 + log(q));
396  }

477 template<typename ELModel>
478 inline void el::InnerELC<ELModel>::lambdaNR(int& nIter, double& maxErr) {
479  // prevent bad starting values
480  lambdaOld_ = lambda0_;
481  lambdaNew_.fill(0.0);
482 
483  MatrixXd GGtUsed_ = GGt_.block(0,0,nEqs_,nObs2_*nEqs_);
484  MatrixXd GUsed_ = G_.block(0,0,nEqs_,nObs2_);
485  block_outer(GGtUsed_,GUsed_);
486 
487  // To avoid numerical problem
488  VectorXd weights_save = weights_;
489  for (int ll=0;ll<nObs2_; ll++) {
490  if (weights_save(ll) < relTol_*nObs_) weights_save(ll) = 0;
491  }
492 
493  // newton-raphson loop
494  int ii;
495  for(ii=0; ii<maxIter_; ii++) {
496  // Q1 and Q2
497  Glambda_.noalias() = lambdaOld_.transpose() * GUsed_;
498  Glambda_ = weights_.head(nObs2_).sum() + Glambda_.array();
499  Q2_.fill(0.0);
500  for(int jj=0; jj<nObs2_; jj++) {
501  rho_(jj) = logsharp1(Glambda_(jj), weights_(jj));
502 
503  // To avoid numerical problem
504  if (weights_(jj) > relTol_*nObs_) {
505  Q2_ += weights_(jj) * logsharp2(Glambda_(jj), weights_(jj)) * GGtUsed_.block(0,jj*nEqs_,nEqs_,nEqs_);
506  }
507  }
508 
509  // To avoid numerical problem
510  for (int kk=0; kk<nObs2_; kk++) {
511  if (isinf(rho_(kk)) && weights_save(kk) == 0) rho_(kk) = 0;
512  }
513 
514  Q1_ = GUsed_ * (rho_.array()*weights_save.head(nObs2_).array()).matrix();
515  // update lambda
516  Q2ldlt_.compute(Q2_);
517  lambdaNew_.noalias() = lambdaOld_ - Q2ldlt_.solve(Q1_);
518  // maxErr = maxRelErr(lambdaNew_, lambdaOld_); // maximum relative error
519  maxErr = maxRelErr();
520  if (maxErr < relTol_) {
521  break;
522  }
523  lambdaOld_ = lambdaNew_; // complete cycle
524  }
525  nIter = ii; // output lambda and also nIter and maxErr
526  return;
527 }
