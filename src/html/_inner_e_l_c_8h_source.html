471 template<typename ELModel>
472 inline void el::InnerELC<ELModel>::lambdaNR(int& nIter, double& maxErr) {
473 
474  lambdaOld_ = lambda0_;
475  lambdaNew_.fill(0.0);
476 
477  // Note: these two cannot be preallocate untill `support` is set
478  // but has to do it this way since .block does not return a MatrixXd (TODO: alternative way?)
479  GGtUsed_ = GGt_.block(0,0,nEqs_,nObs2_*nEqs_);
480  GUsed_ = G_.block(0,0,nEqs_,nObs2_);
481 
482  block_outer(GGtUsed_,GUsed_);
483 
484  // To avoid numerical problem (TODO: test this again and see if it is ok without it?)
485  VectorXd weights_save = weights_;
486  for (int ll=0;ll<nObs2_; ll++) {
487  if (weights_save(ll) < relTol_*nObs_) weights_save(ll) = 0;
488  }
489 
490  // newton-raphson loop
491  int ii;
492  for(ii=0; ii<maxIter_; ii++) {
493  // Q1 and Q2
494  Glambda_.noalias() = lambdaOld_.transpose() * GUsed_;
495  Glambda_ = weights_.head(nObs2_).sum() + Glambda_.array();
496  Q2_.fill(0.0);
497  for(int jj=0; jj<nObs2_; jj++) {
498  rho_(jj) = logsharp1(Glambda_(jj), weights_(jj));
499 
500  // To avoid numerical problem
501  if (weights_(jj) > relTol_*nObs_) {
502  Q2_ += weights_(jj) * logsharp2(Glambda_(jj), weights_(jj)) * GGtUsed_.block(0,jj*nEqs_,nEqs_,nEqs_);
503  }
504  }
505  // To avoid numerical problem
506  for (int kk=0; kk<nObs2_; kk++) {
507  if (isinf(rho_(kk)) && weights_save(kk) == 0) rho_(kk) = 0;
508  }
509  // update lambda
510  Q1_ = GUsed_ * (rho_.array()*weights_save.head(nObs2_).array()).matrix();
511  Q2ldlt_.compute(Q2_);
512  lambdaNew_.noalias() = lambdaOld_ - Q2ldlt_.solve(Q1_);
513  maxErr = maxRelErr();
514  if (maxErr < relTol_) {
515  break;
516  }
517  lambdaOld_ = lambdaNew_; // complete cycle
518  }
519  nIter = ii; // output lambda and also nIter and maxErr
520  return;
521 }
