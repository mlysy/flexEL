229  // stack multiple "Gs" for each quantile level

231  for(int ii=0; ii<y_.size(); ii++) {

252  // TODO: warning if negative sig2?

254  eZg_.array() = (-gamma.transpose()*Z_).array().exp();
255  yXbeZg_.array() = (y_.transpose()-beta.transpose()*X_).array() * eZg_.array();
256  yXbeZg2_.array() = yXbeZg_.array()*yXbeZg_.array();
257 
258  tG_ = MatrixXd::Zero(n_obs_,n_eqs_); // NEW: DEC 25
259  // 1st deriv w.r.t beta
260  tG_.block(0,0,n_obs_,n_bet_) = X_.transpose();
261  tG_.block(0,0,n_obs_,n_bet_).array().colwise() *= yXbeZg_.transpose().array() * eZg_.transpose().array();
262  // 1st deriv w.r.t gamma
263  tG_.block(0,n_bet_,n_obs_,n_gam_) = Z_.transpose();
264  tG_.block(0,n_bet_,n_obs_,n_gam_).array().colwise() *= (1.0-yXbeZg2_.transpose().array()/sig2); // NEW (DEC 6)
265  // variance param
266  tG_.block(0,n_bet_+n_gam_,n_obs_,1).array() = 1/sig2*yXbeZg2_.transpose().array()-1;
267  // quantile param(s)
268  for (int ii=0; ii<n_obs_; ii++) {
269  for (int jj=0; jj<n_qts_; jj++) {
270  tG_.block(ii,n_bet_+n_gam_+1+jj,1,1).array() = phi_tau(yXbeZg_(ii)/sqrt(sig2)-Nu(jj), tau[jj]);
